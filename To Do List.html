import tkinter as tk def add_task(): task = entry.get() if task: listbox.insert(tk.END, task) entry.delete(0, tk.END) def remove_task(): selected_task = listbox.curselection() if selected_task: listbox.delete(selected_task) # Create main window root = tk.Tk() root.title("To-Do List") # Create and pack the listbox to display tasks listbox = tk.Listbox(root, width=50, height=15) listbox.pack(pady=10) # Create and pack the entry widget to add tasks entry = tk.Entry(root, width=50) entry.pack(pady=5) # Create and pack the Add Task button add_button = tk.Button(root, text="Add Task", width=48, command=add_task) add_button.pack() # Create and pack the Remove Task button remove_button = tk.Button(root, text="Remove Task", width=48, command=remove_task) remove_button.pack(pady=5)
